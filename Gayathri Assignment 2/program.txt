import random
while(True):
    temp=random.randint(10,100)
    humidity=random.randint(10,100)
    print("Current Temperature is ",temp)
    print("Current Humidity is ",humidity,"%")
    temperature_ref=37
    humidity_ref=35
    if temp>temperature_ref and humidity<humidity_ref:
        print("Alarm is ON")
    else:
        print("Alarm is OFF")
        break
